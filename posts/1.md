Odin은 C의 대안으로 설계된 시스템 프로그래밍 언어입니다. 간결한 문법과 강력한 기능을 제공하는 Odin의 기초를 알아봅니다.

## Odin이란?

Odin은 게임 개발자 ginger bill이 만든 시스템 프로그래밍 언어로, C의 단점을 보완하면서도 저수준 제어를 유지합니다. "Joy of Programming"을 목표로 설계되었습니다.

## 기본 문법

### Hello World

```odin
package main

import "core:fmt"

main :: proc() {
    fmt.println("Hello, Odin!")
}
```

### 변수와 상수

```odin
package main

import "core:fmt"

main :: proc() {
    // 변수 선언
    x: int = 42
    y := 3.14  // 타입 추론
    
    // 상수
    PI :: 3.14159
    MAX_SIZE :: 100
    
    // 여러 변수 동시 선언
    a, b, c := 1, 2, 3
    
    fmt.printf("x = %d, y = %f\n", x, y)
    fmt.printf("a = %d, b = %d, c = %d\n", a, b, c)
}
```

### 프로시저 (함수)

```odin
package main

import "core:fmt"

// 기본 프로시저
add :: proc(a, b: int) -> int {
    return a + b
}

// 다중 반환값
divide :: proc(a, b: int) -> (quotient: int, remainder: int) {
    return a / b, a % b
}

// 기본값이 있는 매개변수
greet :: proc(name: string, greeting := "Hello") {
    fmt.printf("%s, %s!\n", greeting, name)
}

main :: proc() {
    result := add(10, 20)
    fmt.printf("10 + 20 = %d\n", result)
    
    q, r := divide(17, 5)
    fmt.printf("17 / 5 = %d, 나머지 = %d\n", q, r)
    
    greet("Odin")
    greet("World", "Hi")
}
```

### 배열과 슬라이스

```odin
package main

import "core:fmt"

main :: proc() {
    // 고정 크기 배열
    arr: [5]int = {1, 2, 3, 4, 5}
    
    // 슬라이스
    slice := arr[1:4]  // {2, 3, 4}
    
    // 동적 배열
    dyn: [dynamic]int
    append(&dyn, 10)
    append(&dyn, 20)
    append(&dyn, 30)
    defer delete(dyn)  // 메모리 해제
    
    fmt.println("배열:", arr)
    fmt.println("슬라이스:", slice)
    fmt.println("동적 배열:", dyn)
    
    // 배열 순회
    for value, index in arr {
        fmt.printf("arr[%d] = %d\n", index, value)
    }
}
```

### 구조체

```odin
package main

import "core:fmt"

Person :: struct {
    name: string,
    age:  int,
    height: f32,
}

// 메서드처럼 사용
person_introduce :: proc(p: Person) {
    fmt.printf("안녕하세요, 저는 %s이고 %d살입니다.\n", p.name, p.age)
}

main :: proc() {
    // 구조체 초기화
    person := Person{
        name = "홍길동",
        age = 25,
        height = 175.5,
    }
    
    // 또는 순서대로
    person2 := Person{"김철수", 30, 180.0}
    
    person_introduce(person)
    person_introduce(person2)
}
```

### 열거형과 유니온

```odin
package main

import "core:fmt"

// 열거형
Direction :: enum {
    North,
    South,
    East,
    West,
}

// 태그된 유니온
Value :: union {
    int,
    f64,
    string,
}

main :: proc() {
    dir := Direction.North
    
    switch dir {
    case .North: fmt.println("북쪽")
    case .South: fmt.println("남쪽")
    case .East:  fmt.println("동쪽")
    case .West:  fmt.println("서쪽")
    }
    
    // 유니온 사용
    v: Value = 42
    
    switch val in v {
    case int:    fmt.printf("정수: %d\n", val)
    case f64:    fmt.printf("실수: %f\n", val)
    case string: fmt.printf("문자열: %s\n", val)
    }
}
```

### 에러 처리

```odin
package main

import "core:fmt"
import "core:os"

// Odin은 try/catch가 없음 - 다중 반환값 사용
read_file :: proc(path: string) -> (data: []byte, ok: bool) {
    content, success := os.read_entire_file(path)
    if !success {
        return nil, false
    }
    return content, true
}

main :: proc() {
    data, ok := read_file("example.txt")
    
    if !ok {
        fmt.println("파일을 읽을 수 없습니다")
        return
    }
    
    defer delete(data)  // 메모리 해제
    fmt.println("파일 내용:", string(data))
}
```

## 빌드와 실행

```bash
# 빌드
odin build . -out:program

# 실행
./program

# 디버그 모드
odin build . -out:program -debug

# 릴리즈 모드
odin build . -out:program -o:speed
```

Odin은 C의 복잡함 없이 시스템 프로그래밍의 강력함을 제공합니다.
