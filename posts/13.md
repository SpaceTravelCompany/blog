```se

```

- 계층형 네임스페이스

## 7. Context

```se
context GameCtx {//기본적으로 함수 내부로 복사되어 전파됨. 
    dt: f64,
}

context(GameCtx)
fn update() {
    GameCtx.dt = 0.16666 //즉 update 에서 dt 를 바꿔도 밖의 dt 는 바뀌지 않고 유지됨.
}


context(GameCtx) {
    fn render() {

    }
    fn destroy() {

    }
}
```

    }

    a = A(...)//일반 함수 호출처럼도 사용

    }
    )
}



fn A(int[] a,//전달받은 인수) int[] {

    return a// 어떤 값을 바꾸고 싶은 함수를 만들려면 새로 반환한다.

thread_local mem:vkMem

    f64 dt
