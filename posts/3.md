C 언어는 시스템 프로그래밍과 임베디드 개발의 핵심 언어입니다. 기본 문법부터 메모리 관리까지 알아봅니다.

## C 언어란?

C는 1972년 Dennis Ritchie가 개발한 범용 프로그래밍 언어입니다. Unix 운영체제 개발에 사용되었으며, 현재까지도 운영체제, 드라이버, 임베디드 시스템 개발에 널리 사용됩니다.

## 기본 문법

### Hello World

```c
#include <stdio.h>

int main(void) {
    printf("Hello, World!\n");
    return 0;
}
```

### 변수와 자료형

```c
#include <stdio.h>

int main(void) {
    // 기본 자료형
    int number = 42;
    float decimal = 3.14f;
    char letter = 'A';
    double precise = 3.14159265359;
    
    // 배열
    int numbers[5] = {1, 2, 3, 4, 5};
    char name[] = "Hello";
    
    printf("정수: %d\n", number);
    printf("실수: %f\n", decimal);
    printf("문자: %c\n", letter);
    
    return 0;
}
```

### 포인터

```c
#include <stdio.h>

int main(void) {
    int value = 100;
    int *ptr = &value;  // 포인터 선언 및 주소 할당
    
    printf("값: %d\n", value);
    printf("주소: %p\n", (void*)ptr);
    printf("포인터를 통한 값: %d\n", *ptr);
    
    *ptr = 200;  // 포인터를 통한 값 변경
    printf("변경된 값: %d\n", value);
    
    return 0;
}
```

### 구조체

```c
#include <stdio.h>
#include <string.h>

typedef struct {
    char name[50];
    int age;
    float height;
} Person;

int main(void) {
    Person person;
    strcpy(person.name, "홍길동");
    person.age = 25;
    person.height = 175.5f;
    
    printf("이름: %s\n", person.name);
    printf("나이: %d\n", person.age);
    printf("키: %.1f\n", person.height);
    
    return 0;
}
```

## 메모리 관리

```c
#include <stdio.h>
#include <stdlib.h>

int main(void) {
    // 동적 메모리 할당
    int *arr = (int*)malloc(5 * sizeof(int));
    
    if (arr == NULL) {
        printf("메모리 할당 실패\n");
        return 1;
    }
    
    // 배열 사용
    for (int i = 0; i < 5; i++) {
        arr[i] = i * 10;
    }
    
    // 출력
    for (int i = 0; i < 5; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
    
    // 메모리 해제 (중요!)
    free(arr);
    
    return 0;
}
```

## 컴파일과 실행

```bash
# GCC로 컴파일
gcc -o program main.c

# 실행
./program

# 디버그 정보 포함 컴파일
gcc -g -o program main.c

# 최적화 컴파일
gcc -O2 -o program main.c
```

C 언어는 하드웨어에 가까운 저수준 프로그래밍을 가능하게 하며, 다른 언어들의 기반이 됩니다.
